// ============================================================ // GOOGLE APPS SCRIPT — Expense Tracker Bridge v7 // NEW IN v7: forces the Month column to plain-text format so a // value like "Jun-2026" can never be silently auto-converted into // a real Date object by Google Sheets, regardless of locale. // (Also includes v6's proper unique ID column for reliable // update/delete matching — no more "row not found" errors.) // // Your Sheet columns are: // ID | Date | Raw Input | Description | Amount | Category | // Subcategory | Week | Month // // HOW TO USE: // 1. Open your Google Sheet // 2. Extensions → Apps Script // 3. Select all existing code, delete it // 4. Paste this entire file // 5. Deploy → Manage deployments → pencil icon → New version → Deploy // (keeps your existing URL — no changes needed in the web app // setup box, but you DO need to update the HTML file too) // 6. Make sure "Who has access" is set to "Anyone" // // NOTE: if your Sheet's Month column already has old date-object // values mixed in, those existing rows aren't auto-fixed by this // script — see the app's fallback parsing for why that's still OK. // ============================================================ var HEADERS = ["ID", "Date", "Raw Input", "Description", "Amount", "Category", "Subcategory", "Week", "Month"]; function doPost(e) { try { var data = JSON.parse(e.postData.contents); if (data.action === "update") { return handleUpdate(data); } if (data.action === "delete") { return handleDelete(data); } return handleInsert(data); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function handleInsert(data) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(HEADERS); } var id = data.id || generateId(); sheet.appendRow([ id, data.date || "", data.raw || "", (data.matched || data.raw || ""), data.amount || 0, data.category || "", data.sub || "", data.week || "", data.month || "" ]); // Force the Month column (I) to plain text so Sheets never silently // auto-converts a value like "Jun-2026" into a real Date object, // regardless of locale or auto-detection settings. var lastRow = sheet.getLastRow(); sheet.getRange(lastRow, 9).setNumberFormat("@"); return ContentService .createTextOutput(JSON.stringify({ status: "success", id: id })) .setMimeType(ContentService.MimeType.JSON); } function handleUpdate(data) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rowNum = findRowById(sheet, data.id); if (rowNum === -1) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: "ID not found: " + data.id })) .setMimeType(ContentService.MimeType.JSON); } sheet.getRange(rowNum, 4).setValue(data.matched || data.raw || ""); // Description sheet.getRange(rowNum, 6).setValue(data.category || ""); // Category sheet.getRange(rowNum, 7).setValue(data.sub || ""); // Subcategory return ContentService .createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } function handleDelete(data) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rowNum = findRowById(sheet, data.id); if (rowNum === -1) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: "ID not found: " + data.id })) .setMimeType(ContentService.MimeType.JSON); } sheet.deleteRow(rowNum); return ContentService .createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } function findRowById(sheet, id) { var lastRow = sheet.getLastRow(); if (lastRow < 2 || !id) return -1; var ids = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); for (var i = 0; i < ids.length; i++) { if (ids[i][0].toString() === id.toString()) { return i + 2; // +2: array is 0-indexed, sheet rows start at 1, header is row 1 } } return -1; } function generateId() { return "id" + new Date().getTime() + Math.floor(Math.random() * 1000); } function doGet(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var lastRow = sheet.getLastRow(); if (lastRow < 2) { return ContentService .createTextOutput(JSON.stringify({ status: "success", expenses: [] })) .setMimeType(ContentService.MimeType.JSON); } var range = sheet.getRange(2, 1, lastRow - 1, 9).getValues(); var expenses = range.map(function(row) { return { id: row[0] ? row[0].toString() : "", date: row[1], raw: row[2], matched: row[3], amount: row[4], category: row[5], sub: row[6], week: row[7], month: row[8] }; }).filter(function(e) { return e.id; }); // skip any rows without an ID return ContentService .createTextOutput(JSON.stringify({ status: "success", expenses: expenses })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
